\begin{fact}\label{fact:geometric_sum}
    Let $\field$ be a field, $r\in\field$ be a field element, $a\in\N$ be a natural number. Then
    \[
        \sum_{i=0}^{a}r^i:=
        \begin{cases}
        \Big(\frac{1-r^{a+1}}{1-r}\Big) & r\neq 1 \\
        a+1 & r=1
        \end{cases}
    \]
\end{fact}
